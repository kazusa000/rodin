Usage Example

// Homogeneous Dirichlet BC (u = 0 on boundary)
auto bc = DirichletBC(u, Zero());

// Inhomogeneous Dirichlet BC
auto g = [](const Point& p) { return sin(p.x()); };
auto bc = DirichletBC(u, g).on(1);  // On boundary attribute 1
